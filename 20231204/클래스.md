```python
class Calc_class :
    def __init__(self) :
        self.x = 0
        self.y = 0
        
    def plus(self) :
        return self.x + self.y
    
    def minus(self) :
        return self.x - self.y
    
obj1 = Calc_class()
obj2 = Calc_class()
obj1 == obj2 # False
```

```python
obj1 = Calc_class(10, 20)
obj2 = Calc_class(100, 200)
obj1.plus(), obj2.minus() # (30, -100)
```
